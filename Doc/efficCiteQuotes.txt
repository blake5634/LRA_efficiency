@article{fan2022high,
  title={High-efficient internal resonance energy harvesting: Modelling and experimental study},
  author={Fan, Yimin and Ghayesh, Mergen H and Lu, Tien-Fu},
  journal={Mechanical Systems and Signal Processing},
  volume={180},
  pages={109402},
  year={2022},
  publisher={Elsevier}

"In recent research, to settle the narrow operational bandwidth issue in linear energy harvesting techniques (i.e., to utilise a single-mode and tuning approach), broadband techniques are one of the most promising approaches that are capable of obtaining a specific range of natural vibration sources for efficient energy harvesting purpose. In early applications, the general investigations into linear energy harvesters focused on fundamental resonant modes that could only adapt ambient vibration sources in limited working ranges. Later, adjacent resonant modes, which may have the potential to broaden the operational frequency bandwidth, were considered. Multi-degree-of-freedom (MDOF) and multi-mode energy harvesting techniques utilise multiple resonant modes to improve the system of energy harvesting efficiency, and have therefore been introduced by various researchers."
